beginTransaction(); // Current ordre $curStmt = $pdo->prepare("SELECT ordre FROM composer WHERE numero_du_parcours = ? AND numerodecarte = ?"); $curStmt->execute([$id, $numerodecarte]); $cur = $curStmt->fetch(); if (!$cur) { $pdo->rollBack(); header("Location: modifier_parcours.php?id=" . $id); exit; } $currentOrdre = (int)$cur["ordre"]; // Free current row $tmp = $pdo->prepare("UPDATE composer SET ordre = 0 WHERE numero_du_parcours = ? AND numerodecarte = ?"); $tmp->execute([$id, $numerodecarte]); // If another balise has the target ordre, swap it to current ordre $swap = $pdo->prepare("UPDATE composer SET ordre = ? WHERE numero_du_parcours = ? AND ordre = ?"); $swap->execute([$currentOrdre, $id, $ordre]); // Set new ordre $set = $pdo->prepare("UPDATE composer SET ordre = ? WHERE numero_du_parcours = ? AND numerodecarte = ?"); $set->execute([$ordre, $id, $numerodecarte]); $pdo->commit(); } catch (Exception $e) { if ($pdo->inTransaction()) { $pdo->rollBack(); } } header("Location: modifier_parcours.php?id=" . $id); exit;
